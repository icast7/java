006/**
007 * Parameter passing
008 * @author icastillejos
009 */
010public class ParamTest4 {
011        public static void main(String[] args){
012                
013                /* Test1: Methods can't modify numeric parameters*/
014                System.out.println("Testing tripleValue:");
015                double percent = 10;
016                System.out.println("Before: percent="+percent);
017                tripleValue(percent);
018                System.out.println("After: percent="+percent);
019                
020                /* Test2: Methods can change the state of the object parameters */
021                System.out.println("\nTesting tripleSalary:");
022                Employee4 harry = new Employee4("Harry", 50000);
023                System.out.println("Before: salary= " + harry.getSalary());
024                tripleSalary(harry);
025                System.out.println("After: salary = " + harry.getSalary());
026                
027                /* Test3: Methods can't attach new objects to object parameters */
028                System.out.println("\nTesting swap:");
029                Employee4 a = new Employee4("Alice", 70000);
030                Employee4 b = new Employee4("Bob", 60000);
031                System.out.println("Before: a=" + a.getName());
032                System.out.println("Before: b=" + b.getName());
033                swap(a, b);             
034                System.out.println("After: a=" + a.getName());
035                System.out.println("After: b=" + b.getName());          
036        }
037
038        public static void tripleValue(double x) {
039                x = 3*x;
040                System.out.println("End of method: x="+ x);
041        }
042        
043        public static void tripleSalary(Employee4 x){
044                x.raiseSalary(200);
045                System.out.println("End of method: salary=" + x.getSalary());
046        }
047        
048        public static void swap(Employee4 x, Employee4 y){
049                Employee4 temp = x;
050                x = y;
051                y = temp;
052                System.out.println("End of method: x=" + x.getName());
053                System.out.println("End of method: y=" + y.getName());
054        }
055}       
056        class Employee4{
057                private String name;
058                private double salary;
059                
060                public Employee4(String name, double salary){
061                        this.name = name;
062                        this.salary = salary;
063                }
064                
065                public String getName() {
066                        return name;
067                }
068
069                public double getSalary() {
070                        return salary;
071                }
072                
073                public void raiseSalary(double byPercent) {
074                        double raise = salary * byPercent/100;
075                        salary += raise;
076                }
077                
078        }
