001package com.core.ex1.objclass4;
002/**
003 * Demonstrates static methods and fields
004 * 
005 * @author icastillejos
006 */
007public class StaticTests3 {
008        
009        public static void main(String[] args)
010        {
011                Employee3[] staff = new Employee3[3]; 
012        
013                staff[0] = new Employee3("Tom", 40000); 
014                staff[1] = new Employee3("Dick", 60000);
015                staff[2] = new Employee3("Harry", 65000);
016        
017                
018                
019                for (Employee3 e : staff){
020                        e.setId();
021                        System.out.println("name="+e.getName()+",id="+e.getId()+",salary="+e.getSalary());
022                }
023        
024                
025                
026                int n = Employee3.getNextId();//calls the static method
027                System.out.println("Next available id="+n);
028        }
029}
030
031class Employee3 {
032                private static int nextId = 1;
033                
034                private double salary;
035                private String name;
036                private int id;
037        
038                public Employee3(String n, double s){
039                        this.name = n;
040                        this.salary = s;
041                        this.id = 0;
042                }
043                
044                public String getName() {
045                        return name;
046                }
047        
048                public double getSalary() {
049                        return salary;
050                }
051        
052                public int  getId(){
053                        return id;
054                }
055                
056                public void setId(){
057                        id = nextId;
058                        nextId++;
059                }
060                
061                public static int getNextId(){
062                        return nextId;
063                }
064                
065                public static void main(String[] args) {
066                        Employee3 e  = new Employee3("Harry", 50000);
067                        System.out.println(e.getName() + " " + e.getSalary());
068                }       
069}
